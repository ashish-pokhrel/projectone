IF NOT EXISTS (SELECT 1 FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = 'Category')
BEGIN
CREATE TABLE Category (
    Id UNIQUEIDENTIFIER PRIMARY KEY NOT NULL,
    CategoryName NVARCHAR(255) NOT NULL,
    ImagePath NVARCHAR(255),
    ShowInMenu BIT NOT NULL,
    UpdatedOn DATETIMEOFFSET NOT NULL,
    UpdatedBy UNIQUEIDENTIFIER NOT NULL
);
END

IF NOT EXISTS (SELECT 1 FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = 'ImageHub')
BEGIN
CREATE TABLE ImageHub (
    Id UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),
    TableId UNIQUEIDENTIFIER NOT NULL,
    TableName NVARCHAR(75) NOT NULL,
    ImagePath NVARCHAR(255) NOT NULL,
    IsActive BIT DEFAULT 1,
);
END

IF NOT EXISTS (SELECT 1 FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = 'UserLocation')
BEGIN
CREATE TABLE UserLocation (
    Id UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),
    Latitude DECIMAL(9,6),
    Longitude DECIMAL(9,6),
    City NVARCHAR(255),
    Country NVARCHAR(255),
    StateName NVARCHAR(255)
);
END

IF NOT EXISTS (SELECT 1 FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = 'Feed')
BEGIN
CREATE TABLE Feed
(
    Id UNIQUEIDENTIFIER PRIMARY KEY,
    Content NVARCHAR(MAX),
    Tags NVARCHAR(MAX),
    CategoryId UNIQUEIDENTIFIER,
    Title NVARCHAR(MAX),
    LocationId UNIQUEIDENTIFIER,
    TotalLikes BIGINT,
    TotalShares BIGINT,
    IsActive BIT,
    IsDeleted BIT,
    IsEdited BIT,
    UpdatedOn DATETIMEOFFSET,
    UpdatedBy UNIQUEIDENTIFIER,
    CreatedOn DATETIMEOFFSET,
    CreatedBy UNIQUEIDENTIFIER,
    DeletedBy UNIQUEIDENTIFIER,
    DeletedOn DATETIMEOFFSET
);
END
